⚡ Setup Supabase

Get your free Supabase database from https://supabase.com

  1. Create a Supabase project (free account)
  2. In SQL Editor, paste and run this: 
    CREATE TABLE messages (
  id BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  name TEXT NOT NULL,
  content TEXT NOT NULL,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT now()
);

function getMessages() { const stored = localStorage.getItem('chatMessages'); return stored ? JSON.parse(stored) : []; } function saveMessages(messages) { localStorage.setItem('chatMessages', JSON.stringify(messages)); renderMessages(); updateStatus(); } function sendMessage() { const nameInput = document.getElementById('nameInput'); const messageInput = document.getElementById('messageInput'); const name = nameInput.value.trim() || 'Anonymous'; const message = messageInput.value.trim(); if (!message) return; const messages = getMessages(); messages.push({ id: Date.now(), name: name, content: message, user_id: userId, created_at: new Date().toISOString() }); saveMessages(messages); messageInput.value = ''; messageInput.focus(); } function renderMessages() { const messagesDiv = document.getElementById('messages'); const messages = getMessages(); messagesDiv.innerHTML = ''; messages.forEach(msg => { const isOwn = msg.user_id === userId; const msgDiv = document.createElement('div'); msgDiv.className = 'message ' + (isOwn ? 'own' : 'other'); const time = new Date(msg.created_at).toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }); msgDiv.innerHTML = `
${escapeHtml(msg.name)}
${escapeHtml(msg.content)}
${time}
`; messagesDiv.appendChild(msgDiv); }); // Auto-scroll to bottom messagesDiv.scrollTop = messagesDiv.scrollHeight; } function updateStatus() { const messages = getMessages(); document.getElementById('status').textContent = `Messages: ${messages.length} | Your ID: ${userId.substr(0, 8)}`; } function handleKeyPress(event) { if (event.key === 'Enter') { sendMessage(); } } function clearAllMessages() { if (confirm('Delete all messages?
